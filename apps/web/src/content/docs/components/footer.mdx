 * 
 * @usage
 * 
 * ## Structure
 * 
 * A Footer component is composed of the following parts:
 *
 * - `Footer.astro` - The footer container.
 * - `FooterLinks.astro` - Links within the footer.
 *
 * ```astro
 * ---
 * import Footer from '@bejamas/ui/components/Footer.astro';
 * ---
 * 
 * <Footer />
 * 
 * ```
 * 
 */

import { cn } from "../lib/utils";

const { class: className = "", ...props } = Astro.props;

---

<footer class={cn("bg-footer text-footer-foreground w-full", className)} {...props}>
  {Astro.slots.cta && (
    <div class="flex justify-center">
      <slot name="cta" />
    </div>
  )}
  <slot />
</footer>

## Structure

A Footer component is composed of the following parts:

- `Footer.astro` - The footer container.
- `FooterLinks.astro` - Links within the footer.

```astro
---
import Footer from '@bejamas/ui/components/Footer.astro';
---

<Footer />

```
