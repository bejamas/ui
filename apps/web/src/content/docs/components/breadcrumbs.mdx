<DocsTabItem label="Manual">

```astro
---
/**
 * @component Breadcrumbs
 * @description A composable breadcrumbs navigation component for Astro, using slots for items.
 * 
 * @prop {string} [class] - Additional classes for the breadcrumbs container.
 * 
 * @example
 * 
 * <Breadcrumbs>
 *   <li><Button href="/" as="a" variant="link">Home</Button></li>
 *   <li><ChevronRightIcon /></li>
 *   <li><Button href="/docs" as="a" variant="link">Docs</Button></li>
 *   <li>
 *     <Dropdown>
 *       <span slot="trigger">Menu</span>
 *       <ul slot="content">
 *         <li><Button href="#" as="a" variant="ghost" class="w-full justify-start">Item 1</Button></li>
 *         <li><Button href="#" as="a" variant="ghost" class="w-full justify-start">Item 2</Button></li>
 *         <li><Button href="#" as="a" variant="ghost" class="w-full justify-start">Item 3</Button></li>
 *       </ul>
 *     </Dropdown>
 *   </li>
 *   <li><Button as="span" variant="link">Current Page</Button></li>
 * </Breadcrumbs>
 */

import { cn } from "../lib/utils";

const {
  class: className = "",
  ...props
} = Astro.props;

---

<nav
  aria-label="Breadcrumb"
  class={cn("w-full flex items-center text-sm", className)}
  {...props}
>
  <ol class="flex items-center">
    <slot />
  </ol>
</nav>
```
