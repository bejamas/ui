/**
 * @component Select
 * @title Select
 * @description Accessible select with custom styling and indicator.
 * @status stable
 * @since 0.1.0
 * 
 * @example
 * 
 * <div class="flex flex-col gap-2">
 *   <Label for="fruits">Select a fruit</Label>
 *   <Select id="fruits" class="w-[180px]">
 *     <option value="apple">Apple</option>
 *     <option value="banana">Banana</option>
 *     <option value="cherry">Cherry</option>
 *   </Select>
 * </div>
 */

import { ChevronDown } from "@lucide/astro";
import { cva } from "class-variance-authority";
import { cn } from "../lib/utils";

const {
  label = "",
  options = [],
  value = "",
  name = "",
  id = "",
  required = false,
  disabled = false,
  variant = "default",
  optionClass = "",
  class: className = "",
  showCheckmark = true,
  ...rest
} = Astro.props;

const baseClasses =
  "block w-full select appearance-none relative shadow-xs rounded-lg bg-background ps-4 pe-10 py-2 text-sm text-foreground transition-colors focus:outline-none focus:ring-2 focus:ring-ring focus:border-ring disabled:cursor-not-allowed disabled:opacity-50";

const optionBaseClasses =
  "px-3 py-1.5 text-foreground hover:bg-foreground/10 active:bg-neutral active:text-neutral-content disabled:text-foreground/40";

const selectVariants = cva(baseClasses, {
  variants: {
    variant: {
      default: "border border-border",
      ghost: "border-0 bg-transparent shadow-none focus:bg-background",
      neutral: "border border-neutral/50 focus:border-neutral focus:ring-neutral",
    },
  },
  defaultVariants: {
    variant: "default",
  },
});

const selectClasses = cn(selectVariants({ variant }), !showCheckmark && "hide-checkmark");
const optionClasses = cn(optionBaseClasses, optionClass);
const controlId = id || name || `select-${Math.random().toString(36).slice(2, 9)}`;
---

<div class={cn("relative")}>
    <select
      id={controlId}
      name={name}
      value={value}
      required={required}
      disabled={disabled}
      class={cn("select", selectClasses, className)}
      {...rest}
    >
      {options.length
        ? options.map((opt) => {
            const isObject = typeof opt === "object";
            const optionValue = isObject ? opt.value : opt;
            const optionLabel = isObject ? opt.label ?? opt.value : opt;
            const optionDisabled = isObject ? Boolean(opt.disabled) : false;
            return (
              <option
                value={optionValue}
                selected={value === optionValue}
                disabled={optionDisabled}
                class={optionClasses}
              >
                {optionLabel}
              </option>
            );
          })
        : <slot />
      }
    </select>
    <ChevronDown slot="indicator" class="size-4 text-foreground/60 absolute right-3 top-1/2 -translate-y-1/2" />

</div>
